rules_version = '2';

  match /b/{bucket}/o {
    match /meal-images/{imageId} {
      // Allow anyone to read meal images (public URLs)
      allow read: if true;

      // ‚ö†Ô∏è TESTING ONLY: Allow anyone to upload (no authentication)

// Note: Images are stored as WebP format (~60-80KB per image)
// File naming: meal-images/{timestamp}.webp
